# Análisis de secuencias FASTQ
import Bio.SeqIO as bio

def analyze_genome(file_path):
    sequences = list(bio.parse(file_path, "fastq"))
    gc_content = sum(
        (seq.seq.count('G') + seq.seq.count('C')) / len(seq.seq) 
        for seq in sequences
    ) / len(sequences)
    return {"samples": len(sequences), "avg_gc": gc_content}

# Script de escaneo de vulnerabilidades
#!/bin/bash
echo "Iniciando escaneo Nmap..."
nmap -sV -T4 -p- $1 -oN scan_results.txt

# Detección de CVE conocidos
while read -r line; do
    if [[ $line == *"open"* ]]; then
        port=$(echo $line | cut -d'/' -f1)
        service=$(echo $line | awk '{print $3}')
        echo "[!] Potencial vulnerabilidad en puerto $port ($service)"
    fi
done < scan_results.txt
